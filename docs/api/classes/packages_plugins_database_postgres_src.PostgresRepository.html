Class PostgresRepository<E, ID>Abstract

Abstract PostgreSQL Repository

Provides CRUD operations for entities backed by PostgreSQL. Subclasses must implement entity-to-row and row-to-entity conversions.

Example

class UserRepository extends PostgresRepository<User, string> {
  protected tableName = 'users';

  protected extractId(user: User): string {
    return user.id.value;
  }

  protected toDatabaseRow(user: User): DatabaseRow {
    return {
      id: user.id.value,
      email: user.email,
      name: user.name,
      created_at: user.createdAt,
      updated_at: user.updatedAt,
    };
  }

  protected toDomain(row: DatabaseRow): User {
    return new User(
      new UserId(row.id as string),
      row.email as string,
      row.name as string,
      row.created_at as Date,
      row.updated_at as Date
    );
  }
}
