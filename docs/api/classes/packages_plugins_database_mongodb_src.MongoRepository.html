Class MongoRepository<E, ID>Abstract

Abstract MongoDB Repository

Provides CRUD operations for entities backed by MongoDB. Subclasses must implement entity-to-document and document-to-entity conversions.

Example

class UserRepository extends MongoRepository<User, string> {
  protected collectionName = 'users';

  protected extractId(user: User): string {
    return user.id.value;
  }

  protected toDocument(user: User): MongoDocument {
    return {
      _id: user.id.value,
      email: user.email,
      name: user.name,
      createdAt: user.createdAt,
      updatedAt: user.updatedAt,
    };
  }

  protected toDomain(doc: MongoDocument): User {
    return new User(
      new UserId(doc._id as string),
      doc.email as string,
      doc.name as string,
      doc.createdAt as Date,
      doc.updatedAt as Date
    );
  }
}
